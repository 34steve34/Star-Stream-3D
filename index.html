function fireLaser() { if (!ship) return; // Direction ship is pointing const forward = new THREE.Vector3(0, 0, 1); forward.applyQuaternion(ship.quaternion).normalize(); // Spawn at nose — tune 5.0 if needed (4.0–7.0 range usually good) const noseOffset = 5.0; const spawnPos = ship.position.clone().add(forward.clone().multiplyScalar(noseOffset)); // ── Main bullet group ── const bulletGroup = new THREE.Group(); bulletGroup.position.copy(spawnPos); // Tiny solid core (prevents total disappearance) const coreGeo = new THREE.SphereGeometry(0.7, 8, 6); const coreMat = new THREE.MeshBasicMaterial({ color: 0xffffff }); const core = new THREE.Mesh(coreGeo, coreMat); bulletGroup.add(core); // Bright additive glow layer const glowGeo = new THREE.SphereGeometry(1.6, 12, 10); const glowMat = new THREE.MeshBasicMaterial({ color: 0x99ffff, transparent: true, opacity: 1.0, blending: THREE.AdditiveBlending }); const glow = new THREE.Mesh(glowGeo, glowMat); bulletGroup.add(glow); // Light (slightly toned down but still strong) const bulletLight = new THREE.PointLight(0x88ffff, 500, 60); bulletLight.position.set(0, 0, 0); bulletGroup.add(bulletLight); bulletGroup.userData.velocity = forward.multiplyScalar(55); // nice snappy speed bulletGroup.userData.trail = []; scene.add(bulletGroup); bullets.push(bulletGroup); }
